   s.sid,s.serial#,s.sql_id, s.sql_exec_start, st.sql_text 
from V$session_wait w,v$session s, v$sqltext_with_newlines  st
where 
    st.sql_id = s.sql_id and st.piece=0 and w.sid = s.sid  
    and round((last_call_et)/60) > 15
     and w.event not in 
    ('SQL*Net message from client', --'SQL*Net message to client',

         'gcs remote message', 'DIAG idle wait', 'ges remote message',

         )

select 
   s.sid,s.serial#,s.sql_id, s.sql_exec_start, st.sql_text 
from V$session_wait w,v$session s, v$sqltext_with_newlines  st
where 
    st.sql_id = s.sql_id and st.piece=0 and w.sid = s.sid  
    and round((last_call_et)/60) > 15
     and w.event not in 
    ('SQL*Net message from client', --'SQL*Net message to client',
          'rdbms ipc message',
         'gcs remote message', 'DIAG idle wait', 'ges remote message',
         'pmon timer', 'smon timer',
         'EMON idle wait',

         )
order by s.sql_id
